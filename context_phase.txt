## High-level decisions already made (do not re-debate)

- Public endpoint: `POST /api/leads/contact`. Anyone can submit.
- Anti-spam layers: rate limit (5 req/IP/hour) + honeypot field + structure that allows adding captcha later without rewriting.
- Honeypot field name: `website` (a name spam bots will typically auto-fill, that has no legitimate use for a gym contact form). If the field is non-empty in the submitted body, the request is silently accepted (returns 201) but no row is inserted — confuses bots into not retrying.
- `contact_submission` schema includes: status (enum: new/contacted/qualified/closed), assigned_to (FK → personal, nullable, ON DELETE SET NULL), source (optional varchar).
- Lead management endpoints are staff-only (`requireRole(['employee', 'admin'])`).
- IP-based rate limiting: read real client IP via `x-forwarded-for` (Vercel sets this). The Express app must `app.set('trust proxy', 1)` for the limiter to read it correctly.
- No deletion endpoint for contact submissions — they're a record. Staff can change status to `closed` instead.

## Step 1 — Migration M7: `contact_submission` table

### Generate the file

`npm run migrate:create -- create-contact-submission-table`

### up SQL

```sql
CREATE TABLE public.contact_submission (
  id BIGSERIAL PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL,
  email VARCHAR(150) NOT NULL,
  telefono VARCHAR(20) NULL,
  mensaje TEXT NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'new',
  assigned_to VARCHAR(20) NULL
    REFERENCES public.personal(carnet_identidad) ON DELETE SET NULL,
  source VARCHAR(50) NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT contact_submission_email_format
    CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'),
  CONSTRAINT contact_submission_status_valid
    CHECK (status IN ('new', 'contacted', 'qualified', 'closed')),
  CONSTRAINT contact_submission_nombre_length
    CHECK (LENGTH(TRIM(nombre)) >= 2),
  CONSTRAINT contact_submission_mensaje_length
    CHECK (LENGTH(TRIM(mensaje)) >= 10 AND LENGTH(mensaje) <= 2000)
);

-- Index for listing + filtering by status (common admin query)
CREATE INDEX contact_submission_status_created_at_idx
  ON public.contact_submission(status, created_at DESC);

-- Index for "leads assigned to me" queries
CREATE INDEX contact_submission_assigned_to_idx
  ON public.contact_submission(assigned_to)
  WHERE assigned_to IS NOT NULL;

-- Trigger to auto-update updated_at on row update.
-- Reuse the pattern from member_auth (Phase 1c).
CREATE OR REPLACE FUNCTION public.touch_contact_submission_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at := CURRENT_TIMESTAMP;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_contact_submission_touch_updated_at
BEFORE UPDATE ON public.contact_submission
FOR EACH ROW EXECUTE FUNCTION public.touch_contact_submission_updated_at();
```

### down SQL

```sql
DROP TRIGGER IF EXISTS trg_contact_submission_touch_updated_at
  ON public.contact_submission;
DROP FUNCTION IF EXISTS public.touch_contact_submission_updated_at();
DROP INDEX IF EXISTS public.contact_submission_assigned_to_idx;
DROP INDEX IF EXISTS public.contact_submission_status_created_at_idx;
DROP TABLE IF EXISTS public.contact_submission;
```

- Migration file exists with timestamp ordering after M6.
- Both `up` and `down` are present.
- Table has FK to `personal` with `ON DELETE SET NULL`.
- Indexes exist for the two main query patterns.
- Trigger auto-updates `updated_at`.

## Step 3 — Public lead capture endpoint

### Zod schema

Create `lib/validation/leads.js`:

```javascript
const { z } = require('zod');

// Note: the `website` field is the honeypot. Bots fill it in; humans don't.
// We accept it in the schema (to not reject the request outright — bots
// would get a clear "invalid field" error and adapt) but we'll check its
// emptiness in the controller.
const contactSubmissionSchema = z.object({
  nombre: z.string().trim().min(2).max(100),
  email: z.string().trim().toLowerCase().email().max(150),
  telefono: z.string().trim().max(20).optional().or(z.literal('').transform(() => undefined)),
  mensaje: z.string().trim().min(10).max(2000),
  source: z.string().trim().max(50).optional(),
  website: z.string().optional(),  // honeypot — see controller
});

// Schema for staff updating a lead's status or assignment.
const updateLeadSchema = z.object({
  status: z.enum(['new', 'contacted', 'qualified', 'closed']).optional(),
  assigned_to: z.string().trim().min(5).max(20).nullable().optional(),
}).refine(
  data => data.status !== undefined || data.assigned_to !== undefined,
  { message: 'Debe proporcionar al menos un campo a actualizar.' }
);

module.exports = { contactSubmissionSchema, updateLeadSchema };
```

### Model

Create `models/leadsModel.js`:

```javascript
const pool = require('../config/db');

async function createSubmission({ nombre, email, telefono, mensaje, source }) {
  const result = await pool.query(
    `INSERT INTO public.contact_submission
       (nombre, email, telefono, mensaje, source)
     VALUES ($1, $2, $3, $4, $5)
     RETURNING id, created_at`,
    [nombre, email, telefono || null, mensaje, source || null]
  );
  return result.rows[0];
}

async function listSubmissions({ status, assignedTo, limit = 50, offset = 0 }) {
  // Simple filtered list. Defaults to all leads, most recent first.
  const conditions = [];
  const params = [];
  let p = 1;

  if (status) {
    conditions.push(`status = $${p++}`);
    params.push(status);
  }
  if (assignedTo === 'unassigned') {
    conditions.push(`assigned_to IS NULL`);
  } else if (assignedTo) {
    conditions.push(`assigned_to = $${p++}`);
    params.push(assignedTo);
  }

  const where = conditions.length ? `WHERE ${conditions.join(' AND ')}` : '';
  params.push(limit, offset);

  const result = await pool.query(
    `SELECT cs.id, cs.nombre, cs.email, cs.telefono, cs.mensaje,
            cs.status, cs.assigned_to, cs.source,
            cs.created_at, cs.updated_at,
            p.nombre AS assigned_to_nombre, p.apellido AS assigned_to_apellido
     FROM public.contact_submission cs
     LEFT JOIN public.personal p ON p.carnet_identidad = cs.assigned_to
     ${where}
     ORDER BY cs.created_at DESC
     LIMIT $${p++} OFFSET $${p++}`,
    params
  );
  return result.rows;
}

async function findById(id) {
  const result = await pool.query(
    `SELECT cs.*, p.nombre AS assigned_to_nombre, p.apellido AS assigned_to_apellido
     FROM public.contact_submission cs
     LEFT JOIN public.personal p ON p.carnet_identidad = cs.assigned_to
     WHERE cs.id = $1`,
    [id]
  );
  return result.rows[0] || null;
}

// Whitelist-based update — only `status` and `assigned_to` are updatable.
async function update(id, fields) {
  const allowed = ['status', 'assigned_to'];
  const setClauses = [];
  const params = [];
  let p = 1;

  for (const key of allowed) {
    if (key in fields) {
      setClauses.push(`${key} = $${p++}`);
      params.push(fields[key]);
    }
  }

  if (setClauses.length === 0) {
    const err = new Error('No valid update fields');
    err.code = 'INVALID_UPDATE_BODY';
    throw err;
  }

  params.push(id);
  const result = await pool.query(
    `UPDATE public.contact_submission
     SET ${setClauses.join(', ')}
     WHERE id = $${p}
     RETURNING *`,
    params
  );
  return result.rows[0] || null;
}

module.exports = { createSubmission, listSubmissions, findById, update };
```

### Controller

Create `controllers/leadsController.js`:

```javascript
const Leads = require('../models/leadsModel');
const { contactSubmissionSchema, updateLeadSchema } = require('../lib/validation/leads');

/**
 * Public endpoint. Accepts a contact form submission.
 *
 * Anti-spam layers:
 *   1. Rate limiter (middleware): 5/IP/hour
 *   2. Honeypot: `website` field. If non-empty, the request is silently
 *      "accepted" (201) but nothing is inserted. This prevents bots from
 *      detecting they've been caught and rotating their input.
 *   3. Future: captcha hook. To add captcha later, verify the captcha token
 *      here BEFORE the honeypot check (so failed captchas get a clear 400,
 *      and bots that bypass captcha still hit the honeypot).
 */
async function submitContact(req, res) {
  const parsed = contactSubmissionSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({
      message: 'Datos inválidos.',
      issues: parsed.error.issues.map(i => ({ path: i.path, message: i.message })),
    });
  }
  const data = parsed.data;

  // ---- Captcha hook (future) ----
  // if (CAPTCHA_ENABLED) {
  //   const ok = await verifyCaptcha(req.body.captchaToken, req.ip);
  //   if (!ok) return res.status(400).json({ message: 'Captcha inválido.' });
  // }
  // ---- end captcha hook ----

  // Honeypot: if `website` field has content, silently accept without storing.
  // Logged for monitoring, but no response signal that we caught the bot.
  if (data.website && data.website.trim().length > 0) {
    console.warn('[leads] honeypot triggered', {
      ip: req.ip,
      website_value_length: data.website.length,
    });
    return res.status(201).json({ message: 'Gracias. Nos pondremos en contacto pronto.' });
  }

  try {
    const submission = await Leads.createSubmission({
      nombre: data.nombre,
      email: data.email,
      telefono: data.telefono,
      mensaje: data.mensaje,
      source: data.source,
    });
    return res.status(201).json({
      message: 'Gracias. Nos pondremos en contacto pronto.',
      // Note: deliberately not returning submission.id to the public.
      // Reduces information leakage; staff can find leads via the admin UI.
    });
  } catch (err) {
    console.error('Error en submitContact:', err);
    // Generic message — don't leak DB error details to anonymous callers.
    return res.status(500).json({ message: 'Error interno del servidor.' });
  }
}

/**
 * Staff endpoint. List contact submissions, optional filtering.
 * Query params: status, assigned_to (or 'unassigned'), limit, offset.
 */
async function listLeads(req, res) {
  try {
    const { status, assigned_to } = req.query;
    const limit = Math.min(parseInt(req.query.limit, 10) || 50, 200);
    const offset = Math.max(parseInt(req.query.offset, 10) || 0, 0);

    // Validate status if provided.
    if (status && !['new', 'contacted', 'qualified', 'closed'].includes(status)) {
      return res.status(400).json({ message: 'Status inválido.' });
    }

    const rows = await Leads.listSubmissions({
      status,
      assignedTo: assigned_to,
      limit,
      offset,
    });
    return res.status(200).json({ data: rows });
  } catch (err) {
    console.error('Error en listLeads:', err);
    return res.status(500).json({ message: 'Error interno del servidor.' });
  }
}

/**
 * Staff endpoint. Get one lead by id.
 */
async function getLead(req, res) {
  try {
    const id = parseInt(req.params.id, 10);
    if (!Number.isInteger(id) || id <= 0) {
      return res.status(400).json({ message: 'ID inválido.' });
    }
    const row = await Leads.findById(id);
    if (!row) return res.status(404).json({ message: 'Lead no encontrado.' });
    return res.status(200).json({ data: row });
  } catch (err) {
    console.error('Error en getLead:', err);
    return res.status(500).json({ message: 'Error interno del servidor.' });
  }
}

/**
 * Staff endpoint. Update status and/or assigned_to.
 */
async function updateLead(req, res) {
  try {
    const id = parseInt(req.params.id, 10);
    if (!Number.isInteger(id) || id <= 0) {
      return res.status(400).json({ message: 'ID inválido.' });
    }

    const parsed = updateLeadSchema.safeParse(req.body);
    if (!parsed.success) {
      return res.status(400).json({
        message: 'Datos de actualización inválidos.',
        issues: parsed.error.issues.map(i => ({ path: i.path, message: i.message })),
      });
    }

    const updated = await Leads.update(id, parsed.data);
    if (!updated) return res.status(404).json({ message: 'Lead no encontrado.' });
    return res.status(200).json({ message: 'Lead actualizado.', data: updated });
  } catch (err) {
    if (err.code === 'INVALID_UPDATE_BODY') {
      return res.status(400).json({ message: 'No hay campos válidos para actualizar.' });
    }
    // FK violation on assigned_to (the personal row doesn't exist)
    if (err.code === '23503') {
      return res.status(400).json({
        message: 'El empleado asignado no existe.',
      });
    }
    console.error('Error en updateLead:', err);
    return res.status(500).json({ message: 'Error interno del servidor.' });
  }
}

module.exports = { submitContact, listLeads, getLead, updateLead };
```
